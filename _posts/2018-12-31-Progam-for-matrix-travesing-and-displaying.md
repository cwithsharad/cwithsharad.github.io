**PROGRAM CODE**

```c
#include<iostream.h>
#include<conio.h>

void main()
{
clrscr();
int arr[10][10],n,m,i,j;
cout<<"Enter the number of rows: ";
cin>>m;
cout<<"Enter the number of columns: ";
cin>>n;
cout<<"\nEnter "<<m*n<<" values for the Matrix:\n\n";
for(i=0;i<m;i++)
{
for(j=0;j<n;j++)
{
cin>>arr[i][j];
}
}
cout<<"\nTraversing and Displaying the Given Matrix:\n\n";
for(i=0;i<m;i++)
{
for(j=0;j<n;j++)
{
cout<<"\t"<<arr[i][j];
}
cout<<"\n";
}
getch();
}
```
